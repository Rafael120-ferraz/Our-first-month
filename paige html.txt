        const countdownEl = document.getElementById("countdown");
        const loveEl = document.getElementById("loveMessages");

        function updateCountdown() {
            const now = new Date();
            const nextHour = new Date();
            nextHour.setHours(now.getHours() + 1, 0, 0, 0);
            const diff = nextHour - now;

            const minutes = Math.floor((diff % (1000*60*60)) / (1000*60));
            const seconds = Math.floor((diff % (1000*60)) / 1000);

            countdownEl.innerHTML = `Next I love you in: ${minutes}m ${seconds}s`;
        }

        function addLoveMessage() {
            if (loveEl.childElementCount >= 3) {
                loveEl.removeChild(loveEl.firstChild);
            }
            const msg = document.createElement('p');
            msg.textContent = 'I love you ‚ù§Ô∏è';
            msg.className = 'loveMessage';
            loveEl.appendChild(msg);
        }

        let lastHour = new Date().getHours();

        setInterval(() => {
            const now = new Date();
            updateCountdown();

            if (now.getHours() !== lastHour) {
                addLoveMessage();
                lastHour = now.getHours();
            }
        }, 1000);

        updateCountdown();

        // Floating hearts generator
        function createHeart() {
            const heart = document.createElement('div');
            heart.classList.add('heart');
            heart.textContent = 'üíñ';
            heart.style.left = Math.random() * 100 + 'vw';
            heart.style.fontSize = (20 + Math.random() * 20) + 'px';
            document.body.appendChild(heart);

            setTimeout(() => {
                heart.remove();
            }, 5000);
        }
